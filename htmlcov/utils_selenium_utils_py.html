41class AnyEc: 

42 """Use with WebDriverWait to combine expected_conditions 

43 in an OR. 

44 """ 

45 

46 def __init__(self, *args): 

47 self.ecs = args 

48 

49 def __call__(self, driver): 

50 for fn in self.ecs: 

51 try: 

52 if fn(driver): 

53 return True 

54 except: 

55 pass

58def wait_for_element(d, e_id, time=30): 

59 """ 

60 Uses webdriver(d) to wait for page title(title) to become visible 

61 """ 

62 return WebDriverWait(d, time).until(ec.presence_of_element_located((By.ID, e_id)))

80def wait_for_title(d, title, path): 

81 """ 

82 Uses webdriver(d) to navigate to get(path) until it equals title(title) 

83 """ 

84 while d.title != title:

96def wait_for_either_title(d, title1, title2, time=30): 

97 """ 

98 Uses webdriver(d) to wait for page title(title1 or title2) to become visible 

99 """ 

100 try: 

101 WebDriverWait(d, time).until(AnyEc(ec.title_is(title1), ec.title_is(title2))) 

102 except Exception:

106def wait_for_any_title(d, titles, time=30): 

107 """ 

108 Uses webdriver(d) to wait for page title(any in the list of titles) to become visible 

109 """ 

110 WebDriverWait(d, time).until(AnyEc(*[ec.title_is(title) for title in titles]))

113def button_click_using_xpath(d, xpath): 

114 """ 

115 Uses webdriver(d) to click a button using an XPath(xpath) 

116 """ 

117 button_menu = WebDriverWait(d, 10).until( 

118 ec.element_to_be_clickable((By.XPATH, xpath)) 

119 ) 

120 action = ActionChains(d) 

121 action.move_to_element(button_menu).pause(1).click().perform()

44# object. 

45# 

46# For example, users should be able to do things like

48# omd = omdict([(1,1), (1,11)]) 

49# omd.values(1).append('sup') 

50# omd.allitems() == [(1,1), (1,11), (1,'sup')] 

51# omd.values(1).remove(11) 

52# omd.allitems() == [(1,1), (1,'sup')] 

53# omd.values(1).extend(['two', 'more']) 

54# omd.allitems() == [(1,1), (1,'sup'), (1,'two'), (1,'more')] 

55# 

56# or 

57# 

58# omd = omdict([(1,1), (1,11)]) 

59# omd.allitems().extend([(2,2), (2,22)]) 

60# omd.allitems() == [(1,1), (1,11), (2,2), (2,22)]) 

61# 

62# or 

63# 

64# omd = omdict() 

65# omd.values(1) = [1, 11] 

66# omd.allitems() == [(1,1), (1,11)] 

67# omd.values(1) = list(map(lambda i: i * -10, omd.values(1))) 

68# omd.allitems() == [(1,-10), (1,-110)] 

69# omd.allitems() = filter(lambda (k,v): v > -100, omd.allitems()) 

70# omd.allitems() == [(1,-10)] 

71# 

72# etc. 

73# 

74# To accomplish this, subclass list in such a manner that each list element is 

75# really a two tuple, where the first tuple value is the actual value and the 

76# second tuple value is a reference to the itemlist node for that value. Users 

77# only interact with the first tuple values, the actual values, but behind the 

78# scenes when an element is modified, deleted, inserted, etc, the according 

79# itemlist nodes are modified, deleted, inserted, etc accordingly. In this 

80# manner, users can manipulate omdict objects directly through direct list 

81# manipulation. 

82# 

83# Once accomplished, some methods become redundant and should be removed in 

84# favor of the more intuitive direct value list manipulation. Such redundant 

85# methods include getlist() (removed in favor of values()?), addlist(), and 

86# setlist(). 

87# 

88# With the removal of many of the 'list' methods, think about renaming all 

89# remaining 'list' methods to 'values' methods, like poplist() -> popvalues(), 

90# poplistitem() -> popvaluesitem(), etc. This would be an easy switch for most 

91# methods, but wouldn't fit others so well. For example, iterlists() would 

92# become itervalues(), a name extremely similar to iterallvalues() but quite 

93# different in function.

97class omdict(MutableMapping): 

98 

99 """ 

100 Ordered Multivalue Dictionary. 

101 

102 A multivalue dictionary is a dictionary that can store multiple values per 

103 key. An ordered multivalue dictionary is a multivalue dictionary that 

104 retains the order of insertions and deletions. 

105 

106 Internally, items are stored in a doubly linked list, self._items. A 

107 dictionary, self._map, is also maintained and stores an ordered list of 

108 linked list node references, one for each value associated with that key. 

109 

110 Standard dict methods interact with the first value associated with a given 

111 key. This means that omdict retains method parity with dict, and a dict 

112 object can be replaced with an omdict object and all interaction will 

113 behave identically. All dict methods that retain parity with omdict are: 

114 

115 get(), setdefault(), pop(), popitem(), 

116 clear(), copy(), update(), fromkeys(), len() 

117 __getitem__(), __setitem__(), __delitem__(), __contains__(), 

118 items(), keys(), values(), iteritems(), iterkeys(), itervalues(), 

119 

120 Optional parameters have been added to some dict methods, but because the 

121 added parameters are optional, existing use remains unaffected. An optional 

122 <key> parameter has been added to these methods: 

123 

124 items(), values(), iteritems(), itervalues() 

125 

126 New methods have also been added to omdict. Methods with 'list' in their 

127 name interact with lists of values, and methods with 'all' in their name 

128 interact with all items in the dictionary, including multiple items with 

129 the same key. 

130 

131 The new omdict methods are: 

132 

133 load(), size(), reverse(), 

134 getlist(), add(), addlist(), set(), setlist(), setdefaultlist(), 

135 poplist(), popvalue(), popvalues(), popitem(), poplistitem(), 

136 allitems(), allkeys(), allvalues(), lists(), listitems(), 

137 iterallitems(), iterallkeys(), iterallvalues(), iterlists(), 

138 iterlistitems() 

139 

140 Explanations and examples of the new methods above can be found in the 

141 function comments below and online at 

142 

143 https://github.com/gruns/orderedmultidict 

144 

145 Additional omdict information and documentation can also be found at the 

146 above url. 

147 """

221 def _update_updateall(self, replace_at_most_one, *args, **kwargs): 

222 # Bin the items in <args> and <kwargs> into <replacements> or 

223 # <leftovers>. Items in <replacements> are new values to replace old 

224 # values for a given key, and items in <leftovers> are new items to be 

225 # added. 

226 replacements, leftovers = dict(), [] 

227 for mapping in chain(args, [kwargs]): 

228 self._bin_update_items( 

229 self._items_iterator(mapping), replace_at_most_one, 

230 replacements, leftovers) 

231 

232 # First, replace existing values for each key. 

233 for key, values in six.iteritems(replacements): 

234 self.setlist(key, values) 

235 # Then, add the leftover items to the end of the list of all items. 

236 for key, value in leftovers: 

237 self.add(key, value)

239 def _bin_update_items(self, items, replace_at_most_one, 

240 replacements, leftovers): 

241 """ 

242 <replacements and <leftovers> are modified directly, ala pass by 

243 reference. 

244 """ 

245 for key, value in items: 

246 # If there are existing items with key <key> that have yet to be 

247 # marked for replacement, mark that item's value to be replaced by 

248 # <value> by appending it to <replacements>. 

249 if key in self and key not in replacements: 

250 replacements[key] = [value] 

251 elif (key in self and not replace_at_most_one and 

252 len(replacements[key]) < len(self.values(key))): 

253 replacements[key].append(value) 

254 else: 

255 if replace_at_most_one: 

256 replacements[key] = [value] 

257 else: 

258 leftovers.append((key, value))

766 def __eq__(self, other): 

767 if callable_attr(other, 'iterallitems'): 

768 myiter, otheriter = self.iterallitems(), other.iterallitems() 

769 for i1, i2 in zip_longest(myiter, otheriter, fillvalue=_absent): 

770 if i1 != i2 or i1 is _absent or i2 is _absent: 

771 return False 

772 elif not hasattr(other, '__len__') or not hasattr(other, _items_attr): 

773 return False 

774 # Ignore order so we can compare ordered omdicts with unordered dicts. 

775 else: 

776 if len(self) != len(other): 

777 return False 

778 for key, value in six.iteritems(other): 

779 if self.get(key, _absent) != value: 

780 return False 

781 return True

826def _get_items(mapping): 

827 """Find item iterator for an object.""" 

828 names = ('iterallitems', 'allitems', 'iteritems', 'items') 

829 exist = (n for n in names if callable_attr(mapping, n)) 

830 for a in exist:

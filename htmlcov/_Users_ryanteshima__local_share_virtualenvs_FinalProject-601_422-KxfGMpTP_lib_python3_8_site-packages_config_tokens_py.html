786 def get_char(self): 

787 """ 

788 Get the next char from the stream. Update line and column numbers 

789 appropriately. 

790 

791 :return: The next character from the stream. 

792 :rtype: str 

793 """ 

794 if self.pbchars: 

795 t = self.pbchars.pop() 

796 c = t[0] 

797 self.charline = self.lineno = t[1] 

798 self.charcol = self.colno = t[2] 

799 else: 

800 self.charline = self.lineno 

801 self.charcol = self.colno 

802 c = self.stream.read(1) 

803 if c: 

804 if c != '\n': 

805 self.colno += 1 

806 else: 

807 self.lineno += 1 

808 self.colno = 1 

809 return c

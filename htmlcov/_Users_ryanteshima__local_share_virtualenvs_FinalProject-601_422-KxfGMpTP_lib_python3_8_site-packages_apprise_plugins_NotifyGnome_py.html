155 def send(self, body, title='', notify_type=NotifyType.INFO, **kwargs): 

156 """ 

157 Perform Gnome Notification 

158 """ 

159 

160 if not self._enabled: 

161 self.logger.warning( 

162 "Gnome Notifications are not supported by this system.") 

163 return False 

164 

165 try: 

166 # App initialization 

167 Notify.init(self.app_id) 

168 

169 # image path 

170 icon_path = None if not self.include_image \ 

171 else self.image_path(notify_type, extension='.ico') 

172 

173 # Build message body 

174 notification = Notify.Notification.new(body) 

175 

176 # Assign urgency 

177 notification.set_urgency(self.urgency) 

178 

179 # Always call throttle before any remote server i/o is made 

180 self.throttle() 

181 

182 if icon_path: 

183 try: 

184 # Use Pixbuf to create the proper image type 

185 image = GdkPixbuf.Pixbuf.new_from_file(icon_path) 

186 

187 # Associate our image to our notification 

188 notification.set_icon_from_pixbuf(image) 

189 notification.set_image_from_pixbuf(image) 

190 

191 except Exception as e: 

192 self.logger.warning( 

193 "Could not load Gnome notification icon ({}): {}" 

194 .format(icon_path, e)) 

195 

196 notification.show() 

197 self.logger.info('Sent Gnome notification.') 

198 

199 except Exception: 

200 self.logger.warning('Failed to send Gnome notification.') 

201 self.logger.exception('Gnome Exception') 

202 return False

207 """ 

208 Returns the URL built dynamically based on specified arguments. 

209 """ 

210 

211 _map = { 

212 GnomeUrgency.LOW: 'low', 

213 GnomeUrgency.NORMAL: 'normal', 

214 GnomeUrgency.HIGH: 'high', 

215 } 

216 

217 # Define any URL parameters 

218 params = { 

219 'image': 'yes' if self.include_image else 'no', 

220 'urgency': 'normal' if self.urgency not in _map 

221 else _map[self.urgency], 

222 } 

223 

224 # Extend our parameters 

225 params.update(self.url_parameters(privacy=privacy, *args, **kwargs)) 

226 

227 return '{schema}://?{params}'.format( 

228 schema=self.protocol, 

229 params=NotifyGnome.urlencode(params),

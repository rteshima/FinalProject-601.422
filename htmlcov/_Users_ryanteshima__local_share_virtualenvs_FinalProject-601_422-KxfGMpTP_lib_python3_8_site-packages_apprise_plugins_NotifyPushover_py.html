354 def _send(self, payload, attach=None): 

355 """ 

356 Wrapper to the requests (post) object 

357 """ 

358 

359 if isinstance(attach, AttachBase): 

360 # Perform some simple error checking 

361 if not attach: 

362 # We could not access the attachment 

363 self.logger.error( 

364 'Could not access attachment {}.'.format( 

365 attach.url(privacy=True))) 

366 return False 

367 

368 # Perform some basic checks as we want to gracefully skip 

369 # over unsupported mime types. 

370 if not re.match( 

371 self.attach_supported_mime_type, 

372 attach.mimetype, 

373 re.I): 

374 # No problem; we just don't support this attachment 

375 # type; gracefully move along 

376 self.logger.debug( 

377 'Ignored unsupported Pushover attachment ({}): {}' 

378 .format( 

379 attach.mimetype, 

380 attach.url(privacy=True))) 

381 

382 return True 

383 

384 # If we get here, we're dealing with a supported image. 

385 # Verify that the filesize is okay though. 

386 file_size = len(attach) 

387 if not (file_size > 0 

388 and file_size <= self.attach_max_size_bytes): 

389 

390 # File size is no good 

391 self.logger.warning( 

392 'Pushover attachment size ({}B) exceeds limit: {}' 

393 .format(file_size, attach.url(privacy=True))) 

394 

395 return False 

396 

397 self.logger.debug( 

398 'Posting Pushover attachment {}'.format( 

399 attach.url(privacy=True))) 

400 

401 # Default Header 

402 headers = { 

403 'User-Agent': self.app_id, 

404 } 

405 

406 # Authentication 

407 auth = (self.token, '') 

408 

409 # Some default values for our request object to which we'll update 

410 # depending on what our payload is 

411 files = None 

412 

413 self.logger.debug('Pushover POST URL: %s (cert_verify=%r)' % ( 

414 self.notify_url, self.verify_certificate, 

415 )) 

416 self.logger.debug('Pushover Payload: %s' % str(payload)) 

417 

418 # Always call throttle before any remote server i/o is made 

419 self.throttle() 

420 

421 try: 

422 # Open our attachment path if required: 

423 if attach: 

424 files = {'attachment': (attach.name, open(attach.path, 'rb'))} 

425 

426 r = requests.post( 

427 self.notify_url, 

428 data=payload, 

429 headers=headers, 

430 files=files, 

431 auth=auth, 

432 verify=self.verify_certificate, 

433 timeout=self.request_timeout, 

434 ) 

435 

436 if r.status_code != requests.codes.ok: 

437 # We had a problem 

438 status_str = \ 

439 NotifyPushover.http_response_code_lookup( 

440 r.status_code, PUSHOVER_HTTP_ERROR_MAP) 

441 

442 self.logger.warning( 

443 'Failed to send Pushover notification to {}: ' 

444 '{}{}error={}.'.format( 

445 payload['device'], 

446 status_str, 

447 ', ' if status_str else '', 

448 r.status_code)) 

449 

450 self.logger.debug( 

451 'Response Details:\r\n{}'.format(r.content)) 

452 

453 return False 

454 

455 else: 

456 self.logger.info( 

457 'Sent Pushover notification to %s.' % payload['device']) 

458 

459 except requests.RequestException as e: 

460 self.logger.warning( 

461 'A Connection error occurred sending Pushover:%s ' % ( 

462 payload['device']) + 'notification.' 

463 ) 

464 self.logger.debug('Socket Exception: %s' % str(e)) 

465 

466 return False 

467 

468 except (OSError, IOError) as e: 

469 self.logger.warning( 

470 'An I/O error occurred while reading {}.'.format( 

471 attach.name if attach else 'attachment')) 

472 self.logger.debug('I/O Exception: %s' % str(e)) 

473 return False 

474 

475 finally: 

476 # Close our file (if it's open) stored in the second element 

477 # of our files tuple (index 1) 

478 if files: 

479 files['attachment'][1].close()

481 return True

483 def url(self, privacy=False, *args, **kwargs): 

484 """ 

485 Returns the URL built dynamically based on specified arguments. 

486 """ 

487 

488 _map = { 

489 PushoverPriority.LOW: 'low', 

490 PushoverPriority.MODERATE: 'moderate', 

491 PushoverPriority.NORMAL: 'normal', 

492 PushoverPriority.HIGH: 'high', 

493 PushoverPriority.EMERGENCY: 'emergency', 

494 } 

495 

496 # Define any URL parameters 

497 params = { 

498 'priority': 

499 _map[self.template_args['priority']['default']] 

500 if self.priority not in _map else _map[self.priority], 

501 } 

502 

503 # Only add expire and retry for emergency messages, 

504 # pushover ignores for all other priorities 

505 if self.priority == PushoverPriority.EMERGENCY: 

506 params.update({'expire': self.expire, 'retry': self.retry}) 

507 

508 # Extend our parameters 

509 params.update(self.url_parameters(privacy=privacy, *args, **kwargs)) 

510 

511 # Escape our devices 

512 devices = '/'.join([NotifyPushover.quote(x, safe='') 

513 for x in self.targets]) 

514 

515 if devices == PUSHOVER_SEND_TO_ALL: 

516 # keyword is reserved for internal usage only; it's safe to remove 

517 # it from the devices list 

518 devices = '' 

519 

520 return '{schema}://{user_key}@{token}/{devices}/?{params}'.format( 

521 schema=self.secure_protocol, 

522 user_key=self.pprint(self.user_key, privacy, safe=''), 

523 token=self.pprint(self.token, privacy, safe=''), 

524 devices=devices, 

525 params=NotifyPushover.urlencode(params))

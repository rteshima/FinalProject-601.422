34def encrypt(pt, password): 

35 """Encryption function to securely store user credentials, uses ChaCha_Poly1305 

36 with a user defined SCrypt key.""" 

37 salt = get_random_bytes(32) 

38 key = scrypt(password, salt, key_len=32, N=get_scrypt_cost_factor(), r=8, p=1) 

39 nonce = get_random_bytes(12) 

40 cipher = ChaCha20_Poly1305.new(key=key, nonce=nonce) 

41 ct, tag = cipher.encrypt_and_digest(pt) 

42 json_k = ["nonce", "salt", "ct", "tag"] 

43 json_v = [b64encode(x).decode("utf-8") for x in (nonce, salt, ct, tag)] 

44 result = json.dumps(dict(zip(json_k, json_v))) 

45 

46 return result

49def decrypt(ct, password): 

50 """Decryption function to unwrap and return the decrypted creds back to the main thread.""" 

51 try: 

52 b64Ct = json.loads(ct) 

53 json_k = ["nonce", "salt", "ct", "tag"] 

54 json_v = {k: b64decode(b64Ct[k]) for k in json_k} 

55 key = scrypt( 

56 password, json_v["salt"], key_len=32, N=get_scrypt_cost_factor(), r=8, p=1 

57 ) 

58 cipher = ChaCha20_Poly1305.new(key=key, nonce=json_v["nonce"]) 

59 ptData = cipher.decrypt_and_verify(json_v["ct"], json_v["tag"]) 

60 

61 return ptData 

62 except (KeyError, ValueError): 

63 print("Incorrect Password.") 

64 exit(0)

67def create_encrypted_config(data, file_path): 

68 """Creates an encrypted credential file if none exists. Stores results in a 

69 file in the root directory.""" 

70 if isinstance(data, dict): 

71 data = json.dumps(data) 

72 payload = bytes(data, "utf-8") 

73 log.info("Create a password for the credential file") 

74 cpass = stdiomask.getpass(prompt="Credential file password: ", mask="*") 

75 vpass = stdiomask.getpass(prompt="Verify credential file password: ", mask="*") 

76 if cpass == vpass: 

77 result = encrypt(payload, cpass) 

78 with open(file_path, "w") as f: 

79 f.write(result) 

80 log.info("Credentials safely stored.") 

81 else: 

82 print("Password and verify password do not match.") 

83 exit(0)

116def get_scrypt_cost_factor(mem_percentage=0.5): 

117 # Returns scrypt cost factor 'N' param based off of system memory 

118 # Max value is 2 ** 20 

119 mem = math.floor(virtual_memory().total * mem_percentage / 1024) 

120 # Value must be a power of 2 

121 exponent = math.floor(math.log(mem, 2)) 

122 return min(2 ** 20, 2 ** exponent)

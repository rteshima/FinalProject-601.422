31def await_credential_input(): 

32 username = input("Amazon login ID: ") 

33 password = stdiomask.getpass(prompt="Amazon Password: ") 

34 return { 

35 "username": username, 

36 "password": password,

40def get_credentials(credentials_file, encrypted_pass=None): 

41 if os.path.exists(credentials_file): 

42 credential = load_encrypted_config(credentials_file, encrypted_pass) 

43 return credential["username"], credential["password"] 

44 else: 

45 log.info("No credential file found, let's make one") 

46 log.info("NOTE: DO NOT SAVE YOUR CREDENTIALS IN CHROME, CLICK NEVER!") 

47 credential = await_credential_input() 

48 create_encrypted_config(credential, credentials_file) 

49 return credential["username"], credential["password"]

52class GlobalConfig: 

53 def __init__(self) -> None: 

54 super().__init__() 

55 log.info("Initializing Global configuration...") 

56 # Load up the global configuration 

57 # See http://docs.red-dove.com/cfg/python.html#getting-started-with-cfg-in-python for how to use Config 

58 self.global_config = Cfg(GLOBAL_CONFIG_FILE) 

59 self.fairgame_config = self.global_config.get("FAIRGAME") 

60 self.profile_path = None 

61 self.get_browser_profile_path() 

62 

63 def get_amazon_config(self, encryption_pass=None): 

64 log.info("Initializing Amazon configuration...") 

65 # Load up all things Amazon 

66 amazon_config = self.global_config["AMAZON"] 

67 amazon_config["username"], amazon_config["password"] = get_credentials( 

68 AMAZON_CREDENTIAL_FILE, encryption_pass 

69 ) 

70 return amazon_config 

71 

72 def get_fairgame_config(self): 

73 return self.fairgame_config 

74 

75 def get_browser_profile_path(self): 

76 if not self.profile_path: 

77 self.profile_path = os.path.join( 

78 os.path.dirname(os.path.abspath("__file__")), 

79 self.global_config["FAIRGAME"].get("profile_name", ".profile-amz"), 

80 ) 

81 return self.profile_path

83 def get_property(self, property_name): 

84 if property_name not in self.global_config.keys(): # we don't want KeyError 

85 return None # just return None if not found

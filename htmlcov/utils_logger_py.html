20import coloredlogs 

21import logging 

22import os 

23from utils.version import version 

24from logging import handlers 

25 

26FORMAT = "%(asctime)s|{}|%(levelname)s|%(message)s".format(version) 

27 

28LOG_DIR = "logs" 

29LOG_FILE_NAME = "fairgame.log" 

30if not os.path.exists(LOG_DIR): 

31 try: 

32 os.makedirs(LOG_DIR) 

33 except OSError: 

34 raise 

35 

36LOG_FILE_PATH = os.path.join(LOG_DIR, LOG_FILE_NAME) 

37 

38# This check *must* be executed before logging.basicConfig because, at least on Windows, 

39# basicConfig creates a lock on the log file that prevents renaming. Possibly a workaround 

40# but putting this first seems to dodge the issue 

41if os.path.isfile(LOG_FILE_PATH): 

42 # Create a transient handler to do the rollover for us on startup. This won't 

43 # be added to the logger as a handler... just used to roll the log on startup. 

44 rollover_handler = handlers.RotatingFileHandler( 

45 LOG_FILE_PATH, backupCount=10, maxBytes=100 * 1024 * 1024 

46 ) 

47 # Prior log file exists, so roll it to get a clean log for this run 

48 try: 

49 rollover_handler.doRollover() 

50 except Exception: 

51 # Eat it since it's *probably* non-fatal and since we're *probably* still able to log to the prior file 

52 pass 

53 

54logging.basicConfig( 

55 filename=LOG_FILE_PATH, 

56 level=logging.DEBUG, 

57 format=FORMAT, 

58) 

59 

60log = logging.getLogger("fairgame") 

61log.setLevel(logging.DEBUG) 

62 

63LOGLEVEL = os.environ.get("LOGLEVEL", "INFO").upper() 

64stream_handler = logging.StreamHandler() 

65stream_handler.setFormatter(logging.Formatter(FORMAT)) 

66 

67log.addHandler(stream_handler) 

68 

69coloredlogs.install(LOGLEVEL, logger=log, fmt=FORMAT)

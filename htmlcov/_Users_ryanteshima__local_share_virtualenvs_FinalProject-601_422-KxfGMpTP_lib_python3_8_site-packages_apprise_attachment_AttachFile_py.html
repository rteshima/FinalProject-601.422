26import re 

27import os 

28from .AttachBase import AttachBase

33class AttachFile(AttachBase): 

34 """ 

35 A wrapper for File based attachment sources 

36 """ 

37 

38 # The default descriptive name associated with the service 

39 service_name = _('Local File') 

40 

41 # The default protocol 

42 protocol = 'file' 

43 

44 # Content is local to the same location as the apprise instance 

45 # being called (server-side) 

46 location = ContentLocation.LOCAL 

47 

48 def __init__(self, path, **kwargs): 

49 """ 

50 Initialize Local File Attachment Object 

51 

52 """ 

53 super(AttachFile, self).__init__(**kwargs) 

54 

55 # Store path but mark it dirty since we have not performed any 

56 # verification at this point. 

57 self.dirty_path = os.path.expanduser(path) 

58 return 

59 

60 def url(self, privacy=False, *args, **kwargs): 

61 """ 

62 Returns the URL built dynamically based on specified arguments. 

63 """ 

64 

65 # Define any URL parameters 

66 params = {} 

67 

68 if self._mimetype: 

69 # A mime-type was enforced 

70 params['mime'] = self._mimetype 

71 

72 if self._name: 

73 # A name was enforced 

74 params['name'] = self._name 

75 

76 return 'file://{path}{params}'.format( 

77 path=self.quote(self.dirty_path), 

78 params='?{}'.format(self.urlencode(params)) if params else '', 

79 ) 

80 

81 def download(self, **kwargs): 

82 """ 

83 Perform retrieval of our data. 

84 

85 For file base attachments, our data already exists, so we only need to 

86 validate it. 

87 """ 

88 

89 if self.location == ContentLocation.INACCESSIBLE: 

90 # our content is inaccessible 

91 return False 

92 

93 # Ensure any existing content set has been invalidated 

94 self.invalidate() 

95 

96 if not os.path.isfile(self.dirty_path): 

97 return False 

98 

99 if self.max_file_size > 0 and \ 

100 os.path.getsize(self.dirty_path) > self.max_file_size: 

101 

102 # The content to attach is to large 

103 self.logger.error( 

104 'Content exceeds allowable maximum file length ' 

105 '({}KB): {}'.format( 

106 int(self.max_file_size / 1024), self.url(privacy=True))) 

107 

108 # Return False (signifying a failure) 

109 return False 

110 

111 # We're good to go if we get here. Set our minimum requirements of 

112 # a call do download() before returning a success 

113 self.download_path = self.dirty_path 

114 self.detected_name = os.path.basename(self.download_path) 

115 

116 # We don't need to set our self.detected_mimetype as it can be 

117 # pulled at the time it's needed based on the detected_name 

118 return True 

119 

120 @staticmethod 

121 def parse_url(url): 

122 """ 

123 Parses the URL so that we can handle all different file paths 

124 and return it as our path object 

125 

126 """ 

127 

128 results = AttachBase.parse_url(url, verify_host=False) 

129 if not results: 

130 # We're done early; it's not a good URL 

131 return results 

132 

133 match = re.match(r'file://(?P<path>[^?]+)(\?.*)?', url, re.I) 

134 if not match: 

135 return None 

136 

137 results['path'] = AttachFile.unquote(match.group('path')) 

138 return results

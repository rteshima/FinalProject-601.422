35class ConfigFile(ConfigBase): 

36 """ 

37 A wrapper for File based configuration sources 

38 """

40 # The default descriptive name associated with the service 

41 service_name = _('Local File')

44 protocol = 'file'

49 def __init__(self, path, **kwargs): 

50 """ 

51 Initialize File Object 

52 

53 headers can be a dictionary of key/value pairs that you want to 

54 additionally include as part of the server headers to post with 

55 

56 """ 

57 super(ConfigFile, self).__init__(**kwargs)

59 # Store our file path as it was set 

60 self.path = os.path.abspath(os.path.expanduser(path)) 

61 

62 # Update the config path to be relative to our file we just loaded 

63 self.config_path = os.path.dirname(self.path)

65 return

67 def url(self, privacy=False, *args, **kwargs): 

68 """ 

69 Returns the URL built dynamically based on specified arguments. 

70 """ 

71 

72 # Prepare our cache value 

73 if isinstance(self.cache, bool) or not self.cache: 

74 cache = 'yes' if self.cache else 'no' 

75 

76 else: 

77 cache = int(self.cache) 

78 

79 # Define any URL parameters 

80 params = { 

81 'encoding': self.encoding, 

82 'cache': cache, 

83 } 

84 

85 if self.config_format: 

86 # A format was enforced; make sure it's passed back with the url 

87 params['format'] = self.config_format 

88 

89 return 'file://{path}{params}'.format( 

90 path=self.quote(self.path), 

91 params='?{}'.format(self.urlencode(params)) if params else '', 

92 )

94 def read(self, **kwargs): 

95 """ 

96 Perform retrieval of the configuration based on the specified request 

97 """ 

98 

99 response = None 

100 

101 try: 

102 if self.max_buffer_size > 0 and \ 

103 os.path.getsize(self.path) > self.max_buffer_size: 

104 

105 # Content exceeds maximum buffer size 

106 self.logger.error( 

107 'File size exceeds maximum allowable buffer length' 

108 ' ({}KB).'.format(int(self.max_buffer_size / 1024))) 

109 return None 

110 

111 except OSError: 

112 # getsize() can throw this acception if the file is missing 

113 # and or simply isn't accessible 

114 self.logger.error( 

115 'File is not accessible: {}'.format(self.path)) 

116 return None 

117 

118 # Always call throttle before any server i/o is made 

119 self.throttle() 

120 

121 try: 

122 # Python 3 just supports open(), however to remain compatible with 

123 # Python 2, we use the io module 

124 with io.open(self.path, "rt", encoding=self.encoding) as f: 

125 # Store our content for parsing 

126 response = f.read() 

127 

128 except (ValueError, UnicodeDecodeError): 

129 # A result of our strict encoding check; if we receive this 

130 # then the file we're opening is not something we can 

131 # understand the encoding of.. 

132 

133 self.logger.error( 

134 'File not using expected encoding ({}) : {}'.format( 

135 self.encoding, self.path)) 

136 return None 

137 

138 except (IOError, OSError): 

139 # IOError is present for backwards compatibility with Python 

140 # versions older then 3.3. >= 3.3 throw OSError now. 

141 

142 # Could not open and/or read the file; this is not a problem since 

143 # we scan a lot of default paths. 

144 self.logger.error( 

145 'File can not be opened for read: {}'.format(self.path)) 

146 return None 

147 

148 # Detect config format based on file extension if it isn't already 

149 # enforced 

150 if self.config_format is None and \ 

151 re.match(r'^.*\.ya?ml\s*$', self.path, re.I) is not None: 

152 

153 # YAML Filename Detected 

154 self.default_config_format = ConfigFormat.YAML 

155 

156 # Return our response object 

157 return response

159 @staticmethod 

160 def parse_url(url): 

161 """ 

162 Parses the URL so that we can handle all different file paths 

163 and return it as our path object 

164 

165 """ 

166 

167 results = ConfigBase.parse_url(url, verify_host=False) 

168 if not results: 

169 # We're done early; it's not a good URL 

170 return results 

171 

172 match = re.match(r'[a-z0-9]+://(?P<path>[^?]+)(\?.*)?', url, re.I) 

173 if not match: 

174 return None 

175 

176 results['path'] = ConfigFile.unquote(match.group('path')) 

177 return results
